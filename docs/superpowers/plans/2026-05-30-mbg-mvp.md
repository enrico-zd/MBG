# MBG (My Brand Gue) MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a responsive web app that generates 30–45s product ad videos by calling PixVerse v6 image-to-video for multiple ≤15s segments, polling status, and stitching segments into a final MP4 (no audio) with mock credits/template packs.

**Architecture:** Next.js web app (UI + internal API) + Postgres for persistence + a separate Node worker process using a Postgres-backed queue. PixVerse integration is server-side only. Image is preprocessed into 9:16 before uploading to PixVerse to bias vertical output.

**Tech Stack:** Next.js (App Router), TypeScript, Postgres, Prisma, Node worker, ffmpeg (server-side), object storage (local filesystem for MVP).

```ts
export async function ensureDir(dirPath: string) {
  await mkdir(dirPath, { recursive: true });

  const dir = path.dirname(filePath);
```

```ts
import sharp from "sharp";

type Derive9x16Options = {
  outputWidth: number;
  outputHeight: number;
};

export async function derive9x16Jpeg(input: Buffer, opts: Derive9x16Options) {
  const base = sharp(input);
  const meta = await base.metadata();

  const outW = opts.outputWidth;
  const outH = opts.outputHeight;

  const background = await base
    .clone()
    .resize(outW, outH, { fit: "cover" })
    .blur(30)
    .jpeg({ quality: 80 })
    .toBuffer();

  const foreground = await base
    .clone()
    .resize(outW, outH, { fit: "contain", background: { r: 0, g: 0, b: 0, alpha: 0 } })
    .png()
    .toBuffer();

  const composite = await sharp(background)
    .composite([{ input: foreground, gravity: "center" }])
    .jpeg({ quality: 85 })
    .toBuffer();

  return {
    buffer: composite,
    inputWidth: meta.width ?? 0,
    inputHeight: meta.height ?? 0,
    outputWidth: outW,
    outputHeight: outH,
  };
}
```

```ts
import { PixVerseV2Response, UploadImageResp, GenerateI2VResp, VideoResultResp } from "./types";

const BASE_URL = "https://app-api.pixverse.ai";

function requireApiKey() {
  const key = process.env.PIXVERSE_API_KEY;
  if (!key) throw new Error("PIXVERSE_API_KEY is not set");
  return key;
}

export function newTraceId() {
  return crypto.randomUUID();
}

export async function pixverseUploadImageFromUrl(imageUrl: string, traceId: string) {
  const apiKey = requireApiKey();

  const form = new FormData();
  form.set("image_url", imageUrl);

  const res = await fetch(`${BASE_URL}/openapi/v2/image/upload`, {
    method: "POST",
    headers: {
      "API-KEY": apiKey,
      "Ai-trace-id": traceId,
    },
    body: form,
  });

  const json = (await res.json()) as PixVerseV2Response<UploadImageResp>;
  return json;
}

export async function pixverseGenerateI2V(args: {
  traceId: string;

  imgId: number;

    method: "POST",

      "API-KEY": apiKey,
```

```ts
export type PixVerseV6Quality = "360p" | "540p" | "720p" | "1080p";

export function v6CreditsPerSecondNoAudio(quality: PixVerseV6Quality) {
  switch (quality) {
    case "360p":
      return 5;
    case "540p":
      return 7;
    case "720p":
      return 9;
    case "1080p":
      return 18;
  }
}

export function estimateV6NoAudioCredits(args: { quality: PixVerseV6Quality; seconds: number }) {
  return v6CreditsPerSecondNoAudio(args.quality) * args.seconds;
}
```

```ts
export type MbLanguage = "id" | "en";
export type MbTone = "friendly" | "professional" | "energetic";
export type SegmentKind = "hook" | "benefit" | "offer" | "cta";

export type AdJobParams = {
  language: MbLanguage;
  tone: MbTone;
  productName: string;
  benefits: string[];
  offer: string;
  cta: string;
  targetAudience?: string;
};

export type TemplatePack = {
  id: string;
  name: string;
  baseStyle: string;
};

export function buildSegmentPrompt(args: {
  pack: TemplatePack;
  params: AdJobParams;
  kind: SegmentKind;
  durationSec: number;
}) {
  const langLine = args.params.language === "id" ? "Bahasa Indonesia." : "English.";
  const toneLine =
    args.params.tone === "friendly"
      ? "Tone: friendly, conversational."
      : args.params.tone === "professional"
      ? "Tone: professional, clear."
      : "Tone: energetic, upbeat.";

  const benefits = args.params.benefits.slice(0, 3).map((b) => `- ${b}`).join("\n");

  const kindInstruction =
    args.kind === "hook"
      ? "Create a strong hook in the first 2 seconds. Show the product clearly."
      : args.kind === "benefit"
      ? "Explain 2–3 key benefits with clear product visuals."
      : args.kind === "offer"
      ? "Highlight the offer/promo and urgency."
      : "End with a clear call-to-action.";

  const audience = args.params.targetAudience ? `Target audience: ${args.params.targetAudience}.` : "";

  return [
    `You are creating a vertical 9:16 short-form product ad video segment.`,
    langLine,
    toneLine,
    `Style: ${args.pack.baseStyle}`,
    audience,
    `Product: ${args.params.productName}`,
    `Key benefits:\n${benefits}`,
    `Offer: ${args.params.offer}`,
    `CTA: ${args.params.cta}`,
    `Duration: ${args.durationSec}s`,
    kindInstruction,
    `Avoid text overlays that are unreadable. Keep framing stable and product-centric.`,
  ]
    .filter(Boolean)
    .join("\n");
}
```

```ts
import { NextResponse } from "next/server";
import path from "node:path";
import { prisma } from "../../../../../lib/db/prisma";
import { writeFileToStorage, userProjectDir } from "../../../../../lib/storage/storage";
import { derive9x16Jpeg } from "../../../../../lib/images/derive9x16";

export async function POST(req: Request, ctx: { params: Promise<{ projectId: string }> }) {
  const { projectId } = await ctx.params;
  const form = await req.formData();

  const userId = String(form.get("userId") ?? "");
  const file = form.get("file");
  if (!userId) return NextResponse.json({ error: "userId is required" }, { status: 400 });
  if (!(file instanceof File)) return NextResponse.json({ error: "file is required" }, { status: 400 });

  const buf = Buffer.from(await file.arrayBuffer());
  const assetId = crypto.randomUUID();

  const projectDir = userProjectDir(userId, projectId);
  const originalPath = path.join(projectDir, "assets", assetId, `original-${file.name}`);
  await writeFileToStorage(originalPath, buf);

  const derived = await derive9x16Jpeg(buf, { outputWidth: 720, outputHeight: 1280 });
  const derivedPath = path.join(projectDir, "assets", assetId, `derived_9x16.jpg`);
  await writeFileToStorage(derivedPath, derived.buffer);

  const asset = await prisma.asset.create({
    data: {
      id: assetId,
      userId,
      projectId,
      originalPath,
      derived9x16Path: derivedPath,
      mimeType: file.type || "application/octet-stream",
      width: derived.inputWidth,
      height: derived.inputHeight,
      sizeBytes: BigInt(buf.byteLength),
    },
  });

  return NextResponse.json(asset);
}
```

```ts
import { prisma } from "../db/prisma";

export async function leaseOneTask(args: { workerId: string }) {
  const now = new Date();
  const task = await prisma.$transaction(async (tx) => {
    const next = await tx.task.findFirst({
      where: {
        status: "queued",
        runAfter: { lte: now },
        lockedAt: null,
      },
      orderBy: { createdAt: "asc" },
    });
    if (!next) return null;
    return tx.task.update({
      where: { id: next.id },
      data: { status: "running", lockedAt: now, lockedBy: args.workerId },
    });
  });
  return task;
}

export async function markTaskDone(taskId: string) {
  await prisma.task.update({ where: { id: taskId }, data: { status: "done", lockedAt: null, lockedBy: null } });
}

export async function markTaskFailed(taskId: string, err: string) {
  await prisma.task.update({
    where: { id: taskId },
    data: {
      status: "failed",
      lastError: err.slice(0, 5000),
      lockedAt: null,
      lockedBy: null,
    },
  });
}
```

```ts
import { leaseOneTask, markTaskDone, markTaskFailed } from "../lib/queue/workerLease";
import { prisma } from "../lib/db/prisma";
import { handlePixverseUploadImage } from "./handlers/pixverseUploadImage";
import { handlePixverseGenerateSegment } from "./handlers/pixverseGenerateSegment";
import { handlePixversePollSegment } from "./handlers/pixversePollSegment";
import { handleStitchSegments } from "./handlers/stitchSegments";

async function main() {
  const workerId = process.env.WORKER_ID ?? `worker-${process.pid}`;

  // eslint-disable-next-line no-constant-condition
  while (true) {
    const task = await leaseOneTask({ workerId });
    if (!task) {
      await new Promise((r) => setTimeout(r, 1000));
      continue;
    }

    try {
      if (task.type === "pixverse.upload_image") await handlePixverseUploadImage(task.payloadJson as any);
      else if (task.type === "pixverse.generate_segment") await handlePixverseGenerateSegment(task.payloadJson as any);
      else if (task.type === "pixverse.poll_segment") await handlePixversePollSegment(task.payloadJson as any);
      else if (task.type === "video.stitch_segments") await handleStitchSegments(task.payloadJson as any);
      else throw new Error(`Unknown task type: ${task.type}`);

      await markTaskDone(task.id);
    } catch (e) {
      await markTaskFailed(task.id, e instanceof Error ? e.stack ?? e.message : String(e));
    } finally {
      await prisma.$disconnect();
    }
  }
}

void main();
```

```ts
import { prisma } from "../../lib/db/prisma";
import { enqueueTask } from "../../lib/queue/enqueue";
import { newTraceId, pixverseGetVideoResult } from "../../lib/pixverse/client";

const TTL_MS = 20 * 60 * 1000;

function nextDelayMs(attempts: number) {
  if (attempts <= 3) return 2000;
  if (attempts <= 8) return 5000;
  return 10000;
}

export async function handlePixversePollSegment(payload: { segmentId: string }) {
  const seg = await prisma.adJobSegment.findUnique({ where: { id: payload.segmentId }, include: { adJob: true } });
  if (!seg) throw new Error("segment not found");
  if (seg.status !== "polling") return;
  if (!seg.pixverseVideoId) throw new Error("pixverseVideoId missing");

  const startedAt = seg.startedAt ?? seg.createdAt;
  if (Date.now() - startedAt.getTime() > TTL_MS) {
    await prisma.adJobSegment.update({ where: { id: seg.id }, data: { status: "failed", failureCode: "ttl", failureMessage: "Segment TTL exceeded" } });
    await prisma.adJob.update({ where: { id: seg.adJobId }, data: { status: "failed", failureCode: "ttl", failureMessage: "One or more segments timed out" } });
    return;
  }

  const traceId = newTraceId();
  const res = await pixverseGetVideoResult(Number(seg.pixverseVideoId), traceId);
  const status = res?.Resp?.status;
  await prisma.adJobSegment.update({ where: { id: seg.id }, data: { pixverseStatus: status ?? null } });

  if (status === 1) {
    const url = res?.Resp?.url;
    if (!url) throw new Error("PixVerse returned done but missing url");
    await prisma.adJobSegment.update({ where: { id: seg.id }, data: { status: "done", finishedAt: new Date() } });
    const remaining = await prisma.adJobSegment.count({ where: { adJobId: seg.adJobId, status: { not: "done" } } });
    if (remaining === 0) await enqueueTask("video.stitch_segments", { adJobId: seg.adJobId });
    return;
  }

  if (status === 7) {
    await prisma.adJobSegment.update({ where: { id: seg.id }, data: { status: "failed", failureCode: "moderation", failureMessage: "Moderation failed" } });
    await prisma.adJob.update({ where: { id: seg.adJobId }, data: { status: "failed", failureCode: "moderation", failureMessage: "Moderation failed" } });
    return;
  }

  if (status === 8) {
    await prisma.adJobSegment.update({ where: { id: seg.id }, data: { status: "failed", failureCode: "provider_failed", failureMessage: "PixVerse generation failed" } });
    await prisma.adJob.update({ where: { id: seg.adJobId }, data: { status: "failed", failureCode: "provider_failed", failureMessage: "PixVerse generation failed" } });
    return;
  }

  const task = await prisma.task.findFirst({ where: { type: "pixverse.poll_segment", status: "running" }, orderBy: { updatedAt: "desc" } });
  const attempts = (task?.attempts ?? 0) + 1;
  const delay = nextDelayMs(attempts);
  await enqueueTask("pixverse.poll_segment", { segmentId: seg.id }, new Date(Date.now() + delay));
}
```

```ts
async function run(cmd: string, args: string[]) {
  await new Promise<void>((resolve, reject) => {
    const p = spawn(cmd, args, { stdio: "inherit" });
    p.on("exit", (code) => (code === 0 ? resolve() : reject(new Error(`${cmd} exited ${code}`))));
  });
}

export async function stitchMp4Concat(args: { segmentPaths: string[]; outputPath: string }) {
  const dir = path.dirname(args.outputPath);
  await mkdir(dir, { recursive: true });

  const listPath = path.join(dir, `concat-${Date.now()}.txt`);
  const listBody = args.segmentPaths.map((p) => `file '${p.replaceAll("'", "'\\''")}'`).join("\n");
  await writeFile(listPath, listBody, "utf8");

  await run("ffmpeg", ["-y", "-f", "concat", "-safe", "0", "-i", listPath, "-c", "copy", args.outputPath]);
}
```
